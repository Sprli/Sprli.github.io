代码一：按键K1控制LED指示灯的颜色

# Untitled - By: HP - 周日 8月 20 2023

import time, lcd, image
from modules import ybkey, ybrgb

lcd.init()
lcd.clear(lcd.WHITE)
RGB = ybrgb()
Key = ybkey()
mode = 0
while True:
    if Key.is_press():
        time.sleep_ms(20)
        while Key.is_press():
            pass
        mode += 1
        mode %= 8 
    if mode == 0:
        RGB.set(0, 0, 0)
    elif mode == 1:
        RGB.set(0, 0, 1)
    elif mode == 2:
        RGB.set(0, 1, 0)
    elif mode == 3:
        RGB.set(0, 1, 1)
    elif mode == 4:
        RGB.set(1, 0, 0)
    elif mode == 5:
        RGB.set(1, 0, 1)
    elif mode == 6:
        RGB.set(1, 1, 0)
    elif mode == 7:
        RGB.set(1, 1, 1)
    
    lcd.draw_string(140, 120, "Mode:" + str(mode), lcd.BLACK, lcd.WHITE)
